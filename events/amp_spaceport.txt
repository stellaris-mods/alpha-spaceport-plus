namespace = ampSpaceport

event = {
	id = ampSpaceport.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		set_global_flag = "gf_alphamodplus"
	}
}

planet_event = {
	id = ampSpaceport.200
	hide_window = yes
	trigger = {
		has_spaceport = yes
		has_free_spaceport_module_slot = yes
		switch = {
			trigger = is_spaceport_module_slot_free
			11 = { NOT = { check_variable = { which = "AlphaModSpaceportLevel" value = 11 } } }
			10 = { NOT = { check_variable = { which = "AlphaModSpaceportLevel" value = 10 } } }
			9 = { NOT = { check_variable = { which = "AlphaModSpaceportLevel" value = 9 } } }
			8 = { NOT = { check_variable = { which = "AlphaModSpaceportLevel" value = 8 } } }
			7 = { NOT = { check_variable = { which = "AlphaModSpaceportLevel" value = 7 } } }
		}
	}

	immediate = {
		set_planet_flag = "has_alphamod_upgrade"
		amp_remove_modifiers = yes
		switch = {
			trigger = is_spaceport_module_slot_free
			11 = { amp_buff_11 = yes }
			10 = { amp_buff_10 = yes }
			9 = { amp_buff_9 = yes }
			8 = { amp_buff_8 = yes }
			7 = { amp_buff_7 = yes }
		}
	}
}


planet_event = {
	id = ampSpaceport.300
	hide_window = yes
	trigger = {
		has_planet_flag = "has_alphamod_upgrade"
		has_spaceport = no
	}
	immediate = {
		remove_planet_flag = "has_alphamod_upgrade"
		amp_remove_modifiers = yes
	}
}
